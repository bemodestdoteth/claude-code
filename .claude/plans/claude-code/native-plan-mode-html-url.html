Context

The claude-code-plan OMC skill writes an HTML plan file and shows a reachable hosted URL. Native Plan Mode only shows the markdown plan file path. This change adds HTML companion generation and hosted URL display to native Plan Mode to match that behavior. The preview server already supports the /plan/<project>/<file>.html 3-segment path (scripts/plan-preview.ts:122 — no change needed there).

02

Approach

Add three functions to src/utils/plans.ts: derive project slug from getCwd() basename, compute the HTML plan file path, and write a styled HTML companion file from the approved markdown content. Call the writer from ExitPlanModeV2Tool.ts just before returning the tool result, and include the hosted URL in the result text. Display the URL in UI.tsx alongside the existing plan file path line.

Key code changes

src/utils/plans.ts — new additions after line 143
const HOSTED_PLAN_BASE = 'http://dev.bemodest.me:4173'

export function getProjectSlug(): string {
  return basename(getCwd())
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-|-$/g, '') || 'project'
}

export function getHtmlPlanFilePath(agentId?: AgentId): string {
  const slug = getPlanSlug(getSessionId())
  const dir = join(getPlansDirectory(), getProjectSlug())
  getFsImplementation().mkdirSync(dir)
  const suffix = agentId ? `-agent-${agentId}` : ''
  return join(dir, `${slug}${suffix}.html`)
}

export function getHtmlPlanUrl(agentId?: AgentId): string {
  const slug = getPlanSlug(getSessionId())
  const proj = getProjectSlug()
  const suffix = agentId ? `-agent-${agentId}` : ''
  return `${HOSTED_PLAN_BASE}/plan/${proj}/${slug}${suffix}.html`
}

export async function writeHtmlPlan(
  plan: string,
  agentId?: AgentId
): Promise<string> {
  const filePath = getHtmlPlanFilePath(agentId)
  const html = buildHtmlPlan(plan)
  await writeFile(filePath, html, 'utf-8')
  return filePath
}
